    public sealed record WindowState
    {
        public required WindowKind Kind { get; init; }
        public double? Utilization { get; init; }            // 0..1 ; null si inconnu
        public DateTimeOffset? ResetsAt { get; init; }        // null si inconnu (repli JSONL)
        public double? FractionTimeRemaining { get; init; }   // 0..1 clampé ; null si ResetsAt inconnu
        public long? EstimatedTokens { get; init; }           // somme brute (repli) ; info honnête
        public required SourceReliability Reliability { get; init; }

        /// <summary>Épuisé si utilization connue >= 1. Inconnu (null) != épuisé.</summary>
        public bool Exhausted => Utilization is >= 1.0;

        public static WindowState Unavailable(WindowKind k) =>
            new() { Kind = k, Reliability = SourceReliability.Unavailable };

        /// <summary>Fraction de temps restante clampée [0..1] ; null si reset inconnu ou fenêtre non positive.</summary>
        public static double? FractionRemaining(DateTimeOffset? resetsAt, DateTimeOffset now, TimeSpan windowLength)
        {
            if (resetsAt is null || windowLength <= TimeSpan.Zero) return null;
            var ratio = (resetsAt.Value - now) / windowLength; // TimeSpan / TimeSpan = double
            return Math.Clamp(ratio, 0.0, 1.0);
        }
    }
    ```

    [Fact]
    public void La_couche_Services_ne_reference_aucun_assembly_WPF()
    {
        // Assembly qui contient les types Services/Models (le projet Chronos).
        var asm = typeof(Chronos.Services.IUsageProvider).Assembly;
        string[] interdits = { "PresentationCore", "PresentationFramework", "WindowsBase" };

        // Le test échoue si un provider/modèle tirait une dépendance WPF transitive dans leur graphe direct.
        // On vérifie les types Services/Models un par un : aucun n'expose un type d'un assembly WPF
        // dans ses membres publics (signatures), garantissant la neutralité du contrat.
        var typesData = asm.GetTypes()
            .Where(t => t.Namespace is "Chronos.Services" or "Chronos.Models");

        foreach (var t in typesData)
        {
            var assembliesTouches = t.GetMethods()
                .SelectMany(m => new[] { m.ReturnType }.Concat(m.GetParameters().Select(p => p.ParameterType)))
                .Concat(t.GetProperties().Select(p => p.PropertyType))
                .Select(x => x.Assembly.GetName().Name)
                .Where(n => n is not null);

            Assert.DoesNotContain(assembliesTouches, n => interdits.Contains(n));
        }
    }
    ```

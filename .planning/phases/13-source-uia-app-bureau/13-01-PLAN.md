// src/Chronos/Services/SessionSnapshot.cs (ÉTENDU — ajouts NON cassants en fin de record)
public enum SessionKind { Unknown, Chat, Code, Cowork }
public enum SessionOrigin { Cli, Desktop }
public sealed record SessionSnapshot(
    string SessionId,
    string Project,
    SessionActivity Activity,
    string? Reason,
    System.DateTimeOffset UpdatedAt,
    SessionKind Kind = SessionKind.Unknown,      // défaut Unknown → usages CLI existants inchangés
    SessionOrigin Origin = SessionOrigin.Cli);   // défaut Cli    → usages CLI existants inchangés

// src/Chronos/Services/UiaNode.cs
public sealed record UiaNode(
    string ControlType,
    string Name,
    string? AutomationId,
    bool Enabled,
    System.Collections.Generic.IReadOnlyList<UiaNode> Children);

// src/Chronos/Services/IUiaTreeProvider.cs
public interface IUiaTreeProvider
{
    /// Arbre neutre de la fenêtre Claude, ou null si absente/indisponible. NE LÈVE JAMAIS.
    UiaNode? TryGetTree();
}

// src/Chronos/Services/ISessionSource.cs
public interface ISessionSource
{
    System.Collections.Generic.IReadOnlyList<SessionSnapshot> Read(System.DateTimeOffset now);
}

// src/Chronos/Services/DesktopUiaSessionSource.cs (surface publique)
public sealed class DesktopUiaSessionSource : ISessionSource
{
    public DesktopUiaSessionSource(IUiaTreeProvider provider);
    public void Poll(System.DateTimeOffset now);   // appelé HORS thread UI (Plan 02) — remplit le cache
    public System.Collections.Generic.IReadOnlyList<SessionSnapshot> Read(System.DateTimeOffset now); // rend le cache, non bloquant
    public static System.Collections.Generic.IReadOnlyList<SessionSnapshot> MapTree(UiaNode? root, System.DateTimeOffset now); // PUR, testable
}
